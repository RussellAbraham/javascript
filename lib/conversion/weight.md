### Pounds to Kilograms

`kg=lb/2.2046`

```javascript
function lb2kg(n){
	n = parseFloat(n);
	return n / 2.2046;
};
```

### Pounds to Ounces

```javascript
function lb2oz(n){
	return n * 16;
};
```

function lb2g(n){

function lbs2st(n){
